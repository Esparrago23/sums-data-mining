#   - binario Tesseract-OCR (NO es un paquete pip, pytesseract solo llama al binario)
#     Ubuntu/Debian: sudo apt install tesseract-ocr
#     Windows: winget install --id UB-Mannheim.TesseractOCR -e
#              (agregar "C:\Program Files\Tesseract-OCR" al PATH)
